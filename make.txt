# Makefile for UNIX        

# the compiler: gcc for C program, define as g++ for C++ 
CC = gcc  

# compiler flags: 
# -g enable debugging 
# -Wall turn on all warnings 
# -D__USE_FIXED_PROTOTYPES__  force the compiler to use the correct headers 
# -ansi donâ€™t use GNU extensions, stick to ANSI
# -std=c99 use newer version
CFLAGS=-g -Wall -D__USE_FIXED_PROTOTYPES__ -std=c99 

# the build target executable: 
TARGET = client

all: $(TARGET)

$(TARGET): client.o conn_socket.o connection.o stream.o service.o address_helper.o io_helper.o
	$(CC) $(CFLAGS) -o client main.c client.o conn_socket.o connection.o stream.o service.o address_helper.o io_helper.o

client.o: client.c client.h
	$(CC) $(CFLAGS) -c client.c

conn_socket.o: networking/conn_socket.c networking/conn_socket.h
	$(CC) $(CFLAGS) -c networking/conn_socket.c

connection.o: networking/connection.c networking/connection.h
	$(CC) $(CFLAGS) -c networking/connection.c

stream.o: networking/stream.c networking/stream.h
	$(CC) $(CFLAGS) -c networking/stream.c 

service.o: networking/service.c networking/service.h
	$(CC) $(CFLAGS) -c networking/service.c 

address_helper.o: common/address_helper.c common/address_helper.h
	$(CC) $(CFLAGS) -c common/address_helper.c

io_helper.o: common/io_helper.c common/io_helper.h
	$(CC) $(CFLAGS) -c common/io_helper.c

clean:
	$(RM) $(TARGET) $(TARGET).o
